  <script>
    async function searchPhone() {
      const name = document.getElementById('phoneName').value.trim();
      const resultDiv = document.getElementById('result');
      resultDiv.innerHTML = "جارٍ البحث...";

      try {
        const resp = await fetch(`https://api-mobilespecs.azharimm.dev/v2/search?query=${encodeURIComponent(name)}`);
        const d = await resp.json();
        if (d?.data?.phones?.length > 0) {
          const slug = d.data.phones[0].slug;
          const detail = await (await fetch(`https://api-mobilespecs.azharimm.dev/v2/${slug}`)).json();
          const phone = detail.data;

          let html = `<h2>${phone.phone_name}</h2>`;
          html += `<img src="${phone.thumbnail}" alt="صورة ${phone.phone_name}" width="200"><br>`;
          phone.specifications.forEach(sec => {
            html += `<h3>📌 ${sec.title}</h3><ul>`;
            sec.specs.forEach(s => {
              html += `<li><strong>${s.key}:</strong> ${s.val.join(', ')}</li>`;
            });
            html += `</ul><hr>`;
          });
          resultDiv.innerHTML = html;
        } else {
          resultDiv.innerHTML = "❌ ما تم العثور على جوال بهذا الاسم.";
        }
      } catch {
        resultDiv.innerHTML = "⚠️ حدث خطأ أثناء الاتصال أو جلب البيانات. حاول من جديد.";
      }
    }
  </script>
